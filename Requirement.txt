def create_requirements_txt(library_list, filename="requirements.txt"):
    """
    Creates a requirements.txt file from a list of Python libraries.

    Args:
        library_list: A list of strings, where each string is a Python library name
                      (e.g., ['aiogram', 'requests', 'beautifulsoup4==4.12.0']).
        filename: The name of the file to create (defaults to "requirements.txt").
    """

    try:
        with open(filename, "w") as f:
            for library in library_list:
                f.write(library + "\n")
        print(f"Successfully created {filename} with the specified libraries.")
    except Exception as e:
        print(f"Error creating {filename}: {e}")

# Example Usage:
libraries = ["aiogram==3.3.0", "requests==2.31.0", "beautifulsoup4==4.12.2"] # Replace with your actual list
create_requirements_txt(libraries)
